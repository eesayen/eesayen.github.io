  1. 建立以上对象,有重复之处, 不如定义函数来封装
function account(name,number,balance){
	return{
		name,
		number,
		balance,
		toString(){
			return `${this.name},${this.number},${this.balance}`
		}
	}
}

let acct1=account('justin','1234567',1000);
let acct2=account('Monica','2134567',2000);
  1. 不过以上函数传回的对象,实际上是Object的实例, 有没有办法使传回的对象时Account之类的实例.
function Account(name,number,balances){
	this.name=name;
	this.number=number;
	this.balance=balance;
	this.toString=()=>{

new关键字的意义是建立一个对象作为Account的实例, 再执行Account函数定义的流程, 执行完函数内容后,该实例会作为结果返回
